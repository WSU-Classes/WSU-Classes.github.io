// Round-robin scheduler
int schedule() {
    int start_pid = 0;
    if (prev != 0) {
        start_pid = prev->pid;
    }
    for (int i = 0; i < MAX_PROCS; i++) {
        int pid = (start_pid + i) % MAX_PROCS;
        if (processes[pid].status == PROC_STATUS_READY) {
            next = &processes[pid];
            return 1;
        }
    }
    return 0;
}
